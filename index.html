        return paths[style] || paths[0];

    }


    function gameLoop() {

        const dx = targetPos.x - myData.x;

        const dy = targetPos.y - myData.y;

        if (Math.abs(dx) > 1 || Math.abs(dy) > 1) {

            myData.x += dx * 0.1;

            myData.y += dy * 0.1;

            updatePlayerUI();

        }

        requestAnimationFrame(gameLoop);

    }


    async function init() {

        if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {

            await signInWithCustomToken(auth, __initial_auth_token);

        } else {

            await signInAnonymously(auth);

        }


        onAuthStateChanged(auth, (user) => {

            if (user) {

                myId = user.uid;

                connStatus.innerText = "📍 " + myData.name;

                

                onSnapshot(collection(db, 'artifacts', appId, 'public', 'data', 'players'), (s) => {

                    s.docChanges().forEach(c => {

                        if (c.doc.id === myId) return;

                        if (c.type === 'removed') {

                            document.getElementById('ghost-' + c.doc.id)?.remove();

                        } else {

                            renderEntity(c.doc.id, c.doc.data());

                        }

                    });

                });


                onSnapshot(collection(db, 'artifacts', appId, 'public', 'data', 'graves'), (s) => {

                    graves = {};

                    s.forEach(doc => { graves[doc.id] = doc.data(); });

                    renderGraves();

                }, (err) => console.error("Firestore error:", err));

            }

        });

    }


    window.handleWorldClick = (e) => {

        if (e.target.closest('.ui-panel') || e.target.closest('.create-btn') || e.target.closest('.donate-link')) return;

        const sw = window.innerWidth;

        const sh = window.innerHeight;

        targetPos.x = clamp(e.clientX - (sw/2 - myData.x), BOUNDARY, WORLD_SIZE - BOUNDARY);

        targetPos.y = clamp(e.clientY - (sh/2 - myData.y), BOUNDARY, WORLD_SIZE - BOUNDARY);

    };
